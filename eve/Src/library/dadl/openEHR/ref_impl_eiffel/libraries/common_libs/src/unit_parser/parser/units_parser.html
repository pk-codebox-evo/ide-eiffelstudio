input:
  units
| error
units:
  units_expression
| units_expression annotation
units_expression:
  units_item
| units_expression * units_item
| units_expression / units_item
units_item:
  unit_ref
| unit_ref ^ integer_value
unit_ref:
  V_IDENTIFIER
| V_IDENTIFIER suffix
annotation:
  SYM_START_ANNOT V_IDENTIFIER SYM_END_ANNOT
suffix:
  SYM_START_SUFFIX V_IDENTIFIER SYM_END_SUFFIX
integer_value:
  V_INTEGER
| + V_INTEGER
| - V_INTEGER
